private void checkPauseDetection(double speedKmh, LatLng point) {
    if (speedKmh < PAUSE_SPEED_THRESHOLD) {
        consecutiveSlowPoints++;
        
        // 3 points lents cons√©cutifs = PAUSE
        if (consecutiveSlowPoints >= PAUSE_DETECTION_COUNT && !isPaused) {
            onPauseDetected(point);
        }
    } else {
        if (isPaused) {
            onMovementResumed(point);
        }
        consecutiveSlowPoints = 0;
    }
}

private void onPauseDetected(LatLng point) {
    isPaused = true;
    pauseStartTime = System.currentTimeMillis();
    
    // Marqueur orange de pause
    mMap.addMarker(new MarkerOptions()
        .position(point)
        .title("‚è∏Ô∏è Pause")
        .icon(BitmapDescriptorFactory.defaultMarker(
            BitmapDescriptorFactory.HUE_ORANGE)));
    
    statusText.setText("‚è∏Ô∏è Pause d√©tect√©e");
}

private void onMovementResumed(LatLng point) {
    long pauseDuration = System.currentTimeMillis() - pauseStartTime;
    totalPauseTime += pauseDuration;
    isPaused = false;
    
    // Marqueur cyan de reprise

        .position(point)
